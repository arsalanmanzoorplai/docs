---
title: "Create Search Campaign"
api: "POST /gads/create_search_campaign"
description: "This endpoint is used to create a search campaign."
---

## Request Body

<ParamField body="userId" type="string" required>
	User ID, If not already created, make a user on [create user
	page](/api-reference/endpoint/auth/create_user) first. If already created,
	retrieve the user ID from [get profile
	page](/api-reference/endpoint/auth/get_profile).
</ParamField>

<ParamField body="campaignName" type="string" required>
	Name of the campaign
</ParamField>

<ParamField body="dailyBudget" type="string" required>
	Daily budget of the campaign
</ParamField>

<ParamField body="userListId" type="string | string[]">
	User List ID(s). Can be a single string or an array of strings.
</ParamField>

<ParamField body="startDate" type="string" required>
	Start Date Format: `YYYY-MM-DD`
</ParamField>

<ParamField body="endDate" type="string">
	End Date Format: `YYYY-MM-DD`
</ParamField>

 <ParamField body="keywords" type="object[]" required>
	Keywords for the campaign. You can either type your own keywords or use our [generate keyword ideas API](/api-reference/endpoint/google/generate_keyword_ideas) to get keyword suggestions.
	
	<Expandable title="Keyword Object">
		<ParamField body="text" type="string" required>
			The keyword text
		</ParamField>
		
		<ParamField body="match_type" type="string" required>
			The match type for the keyword

			Valid `match_type` values: `EXACT`, `PHRASE`, `BROAD`
 
		</ParamField>
	</Expandable>
</ParamField> 



<ParamField body="negativeKeywords" type="object[]">
	Negative keywords for the campaign. You can either type your own keywords or use our [generate keyword ideas API](/api-reference/endpoint/google/generate_keyword_ideas) to get keyword suggestions.
	
	<Expandable title="Negative Keyword Object">
		<ParamField body="text" type="string" required>
			The keyword text
		</ParamField>
		
		<ParamField body="match_type" type="string" required>
			The match type for the keyword

			Valid `match_type` values: `EXACT`, `PHRASE`, `BROAD`
 
		</ParamField>
	</Expandable>
</ParamField>

<ParamField body="locations" type="string[]">
	Target locations for the campaign. Array of location IDs. First, use the [get_location_v2 API](/api-reference/endpoint/google/get_location_v2) to search for location names (e.g., "Birmingham, London") and get their corresponding `locationId`. Then pass those `locationId` in this array.
</ParamField>

<ParamField body="proximityTargeting" type="object[]">
	Proximity targeting for the campaign. Array of proximity targeting objects.
	
	<Expandable title="Proximity Targeting Object">
		<ParamField body="radius_units" type="string" required>
			Units for the radius (e.g., "MILES", "KILOMETERS")
		</ParamField>
		
		<ParamField body="radius" type="number" required>
			Radius value for the proximity targeting
		</ParamField>
		
		<ParamField body="longitude_in_micro_degrees" type="number" required>
			Longitude in micro degrees
		</ParamField>
		
		<ParamField body="latitude_in_micro_degrees" type="number" required>
			Latitude in micro degrees
		</ParamField>
	</Expandable>
</ParamField>

		<ParamField body="negativeLocations" type="string[]">
			Negative locations for the campaign. Locations that should be excluded from targeting. Array of location IDs. First, use the [get_location_v2 API](/api-reference/endpoint/google/get_location_v2) to search for location names (e.g., "Birmingham, London") and get their corresponding `locationId`. Then pass those `locationId` in this array.
		</ParamField>

<ParamField body="languages" type="string[]">
	Languages for the campaign. Array of language code strings.
	
		Language code string (e.g., "en", "es", "fr", etc.)
</ParamField>

<ParamField body="ads" type="object[]">
	Ads array for the campaign. Array of ad objects containing creative elements.
	
	<Expandable title="Ads Object">
		<ParamField body="url" type="string" required>
			Destination URL for the ad
		</ParamField>
		
		<ParamField body="headlines" type="string[]" required>
			Array of headline strings for the ad
		</ParamField>
		
		<ParamField body="descriptions" type="string[]" required>
		  	Array of description strings for the ad 
		</ParamField>
		
		<ParamField body="headlineIndexToPinFirstSlot" type="number">
			Index of the headline to pin to the first slot (optional)
		</ParamField>
		
		<ParamField body="descriptionIndexToPinFirstSlot" type="number">
			Index of the description to pin to the first slot (optional)
		</ParamField>
	</Expandable>
</ParamField>

<ParamField body="conversionCategory" type="string[]">
	Conversion categories for the campaign. Array of conversion category strings. First, use the [get_conversion_action API](/api-reference/endpoint/google/get_conversion_action) to retrieve available conversion categories, then pass the desired category strings in this array.
</ParamField>

<ParamField body="callouts" type="string[]">
		Callouts are Keywords for the campaign. You can either type your own keywords or use our [generate keyword ideas API](/api-reference/endpoint/google/generate_keyword_ideas) to get keyword suggestions.
</ParamField>

<ParamField body="phoneNumber" type="string">
	Phone number for the campaign (optional)
</ParamField>

<ParamField body="phoneNumberCountryCode" type="string">
	Country code for the phone number (optional)
</ParamField>

<ParamField body="leadFormAssetId" type="string">
	Lead form asset ID for the campaign (optional). First, use the [get_lead_forms API](/api-reference/endpoint/google/get_lead_forms) to retrieve available lead form asset IDs, then pass the desired asset ID in this field.
</ParamField>

<ParamField body="siteLink" type="object[]">
	Site links array for the campaign. Array of site link objects. Minimum 2 items required.
	
	<Expandable title="Site Link Object">
		<ParamField body="sitelinkText" type="string" required>
			Text for the site link
		</ParamField>
		
		<ParamField body="url" type="string" required>
			URL for the site link
		</ParamField>
		
		<ParamField body="line1" type="string" required>
			First line of the site link description
		</ParamField>
		
		<ParamField body="line2" type="string" required>
			Second line of the site link description
		</ParamField>
	</Expandable>
</ParamField>

<ParamField body="images" type="string[]">
	 Images for the campaign. Array of image URLs.
</ParamField>

<ParamField body="tracking_url_template" type="string">
	Tracking URL template for the campaign (optional)
</ParamField>

<ParamField body="currencyCode" type="string">
	Currency code for the campaign (optional)
</ParamField>

<ParamField body="geo_target_type" type="string">
	Geographic targeting type for the campaign
	
	Valid values: `PRESENCE`, `PRESENCE_OR_INTEREST`
</ParamField>
